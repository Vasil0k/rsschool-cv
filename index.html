function dataReverse(data) {
  const byteArr = [];
  for (let i = 0; i < data.length; i += 8) {
    byteArr.push(data.slice(i, i + 8))
  }
  for (let i = 0; i < Math.floor(byteArr.length / 2); i++) {
    const temp = byteArr[i];
    const lastIdx = byteArr.length - i - 1;
    byteArr[i] = byteArr[lastIdx];
    byteArr[lastIdx] = temp;
  }
  return byteArr.flat();
}
